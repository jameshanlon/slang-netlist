69 auto detectCycles() {
70 std::set<CycleType> cycles;
71
72 // Start a DFS traversal from each node
73 for (const auto &nodePtr : graph) {
74 CycleDetectionVisitor<NodeType, EdgeType> visitor;
75 const auto *startNode = nodePtr.get();
76 if (visitedNodes.count(startNode) == 0) {
77
78 // Mark the starting node as visited.
79 visitedNodes.insert(startNode);
80
81 // Perform DFS traversal.
82 DepthFirstSearch<NodeType, EdgeType,
83 CycleDetectionVisitor<NodeType, EdgeType>,
84 EdgePredicate>
85 dfs(visitor, *nodePtr);
86
87 // Additionally, mark all nodes in cycles as visited to avoid
88 // redundant DFS calls.
89 markAllVisitedNodes(visitor);
90 }
91
92 // Add any cycles that were found.
93 cycles.insert(visitor.cycles.begin(), visitor.cycles.end());
94 }
95
96 // Return a vector.
97 std::vector<CycleType> result(cycles.begin(), cycles.end());
98
99 // Canonicalise the result by sorting by node ID.
100 auto byNodeId = [](CycleType const &a, CycleType const &b) {
101 return std::lexicographical_compare(
102 a.begin(), a.end(), b.begin(), b.end(),
103 [](auto const &pa, auto const &pb) { return pa->ID < pb->ID; });
104 ;
105 };
106 std::sort(result.begin(), result.end(), byNodeId);
107
108 return result;
109 }
