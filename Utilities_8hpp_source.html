55 static bool wildcardMatch(const char *text, const char *pattern) {
56 while (*pattern != '\0') {
57 if (*pattern == '*') {
58 if (wildcardMatch(text, pattern + 1)) {
59 return true;
60 }
61 if (*text != '\0' && wildcardMatch(text + 1, pattern)) {
62 return true;
63 }
64 return false;
65 } else if (*pattern == '?') {
66 if (*text == '\0') {
67 return false;
68 }
69 pattern++;
70 text++;
71 } else {
72 if (*pattern != *text) {
73 return false;
74 }
75 pattern++;
76 text++;
77 }
78 }
79 return *text == '\0' && *pattern == '\0';
80 }

94 static auto formatTable(FormatBuffer &buffer, const Row &header,
95 const Table &rows, TableFormatConfig cfg = {}) {
96
97 const std::size_t cols = header.size();
98
99 // Compute column widths: max of header and all rows for each column.
100 std::vector<std::size_t> widths(cols);
101
102 for (std::size_t col = 0; col < cols; ++col) {
103 std::size_t max_width = header[col].size();
104 for (const auto &row : rows) {
105 if (col < row.size()) {
106 max_width = std::max(max_width, row[col].size());
107 }
108 }
109 widths[col] = max_width;
110 }
111
112 auto appendRow = [&](const Row &row) {
113 for (std::size_t col = 0; col < cols; ++col) {
114 std::string_view value;
115 if (col < row.size()) {
116 value = row[col];
117 } else {
118 value = "";
119 }
120
121 // Left-align in a field of `widths[col]`.
122 buffer.format("{:<{}}", value, widths[col]);
123
124 // Padding between columns (except after the last).
125 if (col + 1 < cols && cfg.padding > 0) {
126 buffer.format("{: >{}}", "", cfg.padding);
127 }
128 }
129 buffer.format("\n");
130 };
131
132 appendRow(header);
133 for (const auto &row : rows) {
134 appendRow(row);
135 }
136 }
