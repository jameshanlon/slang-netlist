127 auto addEdge(NodeType &targetNode) -> EdgeType & {
128 bool isSelfEdge = (&getDerived() == &targetNode);
129 std::lock_guard<std::mutex> lock(edgeMutex);
130 if (auto it = outEdgeIndex.find(&targetNode); it != outEdgeIndex.end()) {
131 return *it->second;
132 }
133 auto edge = std::make_shared<EdgeType>(getDerived(), targetNode);
134 outEdges.emplace_back(edge);
135 outEdgeIndex.emplace(&targetNode, edge.get());
136 if (isSelfEdge) {
137 inEdges.push_back(edge);
138 } else {
139 std::lock_guard<std::mutex> lock2(targetNode.edgeMutex);
140 targetNode.inEdges.push_back(edge);
141 }
142 return *edge.get();
143 }
144
152 auto addNewEdge(NodeType &targetNode) -> EdgeType & {
153 bool isSelfEdge = (&getDerived() == &targetNode);
154 auto edge = std::make_shared<EdgeType>(getDerived(), targetNode);
155 if (isSelfEdge) {
156 std::lock_guard<std::mutex> lock(edgeMutex);
157 outEdges.emplace_back(edge);
158 // If no entry exists yet (caller went straight to addNewEdge), seed
159 // it so a later addEdge dedupes against this edge instead of
160 // creating a third parallel one.
161 outEdgeIndex.try_emplace(&targetNode, edge.get());
162 inEdges.push_back(edge);
163 } else {
164 {
165 std::lock_guard<std::mutex> lock(edgeMutex);
166 outEdges.emplace_back(edge);
167 outEdgeIndex.try_emplace(&targetNode, edge.get());
168 }
169 {
170 std::lock_guard<std::mutex> lock(targetNode.edgeMutex);
171 targetNode.inEdges.push_back(edge);
172 }
173 }
174 return *edge.get();
175 }

179 auto removeEdge(NodeType &targetNode) -> bool {
180 auto edgeIt = findEdgeTo(targetNode);
181 if (edgeIt != outEdges.end()) {
182 auto success = targetNode.removeInEdge(getDerived());
183 assert(success && "No corresponding in edge reference");
184 outEdges.erase(edgeIt);
185 // Re-point or drop the index entry: a parallel edge to the same
186 // target may still survive in outEdges.
187 auto survivor = std::ranges::find_if(outEdges, [&](auto &e) {
188 return &e->getTargetNode() == &targetNode;
189 });
190 if (survivor == outEdges.end()) {
191 outEdgeIndex.erase(&targetNode);
192 } else {
193 outEdgeIndex[&targetNode] = survivor->get();
194 }
195 return success;
196 }
197 return false;
198 }

201 void clearAllEdges() {
202 // Remove outgoing edges.
203 for (auto &edge : outEdges) {
204 edge->getTargetNode().removeInEdge(getDerived());
205 }
206 outEdges.clear();
207 outEdgeIndex.clear();
208 // Remove incoming edges, creating a temporary list to avoid
209 // invalidating the iterator.
210 std::vector<NodeType *> sourceNodes;
211 for (auto &edge : inEdges) {
212 sourceNodes.push_back(&edge->getSourceNode());
213 }
214 for (auto *sourceNode : sourceNodes) {
215 sourceNode->removeEdge(getDerived());
216 }
217 assert(inEdges.empty());
218 }

222 auto getEdgesTo(const NodeType &targetNode, std::vector<EdgeType *> &result)

225 for (auto &edge : outEdges) {
226 if (edge->getTargetNode() == targetNode) {
227 result.push_back(edge.get());
228 }
229 }
230 return !result.empty();
231 }

341 std::lock_guard<std::mutex> lock(nodesMutex);

350 auto removeNode(NodeType &nodeToRemove) -> bool {
351 auto nodeToRemoveDesc = findNode(nodeToRemove);
352 if (nodeToRemoveDesc >= nodes.size()) {
353 // The node is not in the graph.

357 nodeToRemove.clearAllEdges();
358 // Remove the node itself.

360 return true;
